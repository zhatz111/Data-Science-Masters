with open("ZHatzenbeller_chatgpt_response.txt", "r", encoding="utf8") as f:
    file_content = f.read()
    text_ = nltk.word_tokenize(file_content)
    sent_ = nltk.sent_tokenize(file_content)

chatgpt_txt = [i.lower() for i in text_ if i.isalpha()]
chatgpt_sent = [i.lower() for i in sent_ if len(i) > 5]

polarity = []
for sentence in chatgpt_sent:
    sid = SentimentIntensityAnalyzer()
    ss = sid.polarity_scores(sentence)
    polarity.append(ss)
In [176]:
