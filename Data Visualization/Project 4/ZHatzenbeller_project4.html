    sent_ = nltk.sent_tokenize(file_content)

Sentiment Analysis

In[173]:
polarity = []
for sentence in chatgpt_sent:
    sid = SentimentIntensityAnalyzer()
    ss = sid.polarity_scores(sentence)
    polarity.append(ss)
In[176]:
neg = []
neu = []
pos = []
compound = []
for i in polarity:
    neg.append(i["neg"])
    neu.append(i["neu"])
    pos.append(i["pos"])
    compound.append(i["compound"])
