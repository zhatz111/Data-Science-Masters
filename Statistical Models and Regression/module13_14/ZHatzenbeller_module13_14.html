# Define Mitcherlich function: y = theta1 - theta2 * exp(-theta3 * x)
def mitch(x, theta1, theta2, theta3):
    return theta1 - theta2 * np.exp(-theta3 * x)

tup_list = []
for idx, row in sample_data_1211.iterrows():
    for value in row["Available Chlorine, yi"]:
        tup_list.append((row["Time, xi"], value))

tup_list.sort()
xs = np.array([t[0] for t in tup_list])
ys = np.array([t[1] for t in tup_list])

# Starting values reasoning:
# theta1 (asymptote as x -> infinity) ~ smallest observed mean (around 0.39)
# theta2 ~ theta1 - y_at_small_x. Use y at smallest x (x=8) mean ~0.49 so theta2 ~ 0.49 - theta1
# theta3 (rate) choose small positive value; we can estimate roughly by solving for rate from two points.
theta1_start = ys.min() - 0.005  # slightly below smallest observed
y_at_8 = np.mean(np.array(sample_data_1211[sample_data_1211['Time, xi']==8]['Available Chlorine, yi'].values[0]))
theta2_start = (y_at_8 - theta1_start)
# crude rate: using two points (mean at 8 and mean at 40) -> theta3 ~ -log((y40-theta1)/ (y8-theta1)) / (40-8)
y_at_40 = np.mean(np.array(sample_data_1211[sample_data_1211['Time, xi']==40]['Available Chlorine, yi'].values[0]))
# protect against division by zero
if (y_at_8 - theta1_start)==0 or (y_at_40 - theta1_start)==0:
    theta3_start = 0.05
else:
    ratio = max((y_at_40 - theta1_start) / (y_at_8 - theta1_start), 1e-8)
    theta3_start = -np.log(ratio) / (40 - 8)
    if theta3_start <= 0:
        theta3_start = 0.05

p0 = [theta1_start, theta2_start, theta3_start]

# Fit using curve_fit (nonlinear least squares)
popt, pcov = curve_fit(mitch, xs, ys, p0=p0, maxfev=10000)
theta1_hat, theta2_hat, theta3_hat = popt
y_pred = mitch(xs, *popt)

# Standard errors from covariance matrix
se = np.sqrt(np.diag(pcov))
# Degrees of freedom for residuals
n = len(ys)
p = len(popt)
df_resid = n - p

# Calculate residuals and sum of squares
residuals = ys - y_pred
ss_res = np.sum(residuals**2)
ss_tot = np.sum((ys - np.mean(ys))**2)
ss_reg = ss_tot - ss_res

# Calculate p-values for parameters (approximate, using t-distribution)
t_stats = popt / se
p_values = [2 * (1 - stats.t.cdf(np.abs(t), df_resid)) for t in t_stats]

# Approximate 95% CIs using t quantile with df_resid
tval = stats.t.ppf(0.975, df_resid)
ci_lower = popt - tval * se
ci_upper = popt + tval * se

# Overall F-test for regression vs intercept-only (nonlinear: use same idea)
df_reg = p - 1  # parameters beyond intercept-like term; for non-linear this is approximate
ms_reg = ss_reg / df_reg
ms_res = ss_res / df_resid
F_stat = ms_reg / ms_res
p_value_F = 1 - stats.f.cdf(F_stat, df_reg, df_resid)

# Print numeric results and summary table
results = pd.DataFrame({
    'parameter': ['theta1', 'theta2', 'theta3'],
    'estimate': popt,
    'std_error': se,
    'p_value': p_values,
    'CI_lower_95%': ci_lower,
    'CI_upper_95%': ci_upper
})

print("Starting values used (p0):", p0)
print("\nParameter estimates and approx. 95% CIs:")
print(results.to_string(index=False))
print(f"\nResidual degrees of freedom: {df_resid} (n={n}, p={p})")
print(f"Residual sum of squares: {ss_res:.6f}")
print(f"Total sum of squares: {ss_tot:.6f}")
print(f"Regression sum of squares (SS_reg): {ss_reg:.6f}")
print(f"F-statistic (approx): {F_stat:.4f} on ({df_reg},{df_resid}) df, p-value = {p_value_F:.4g}")

x1 = sample_data_1216["x1 (% NG)"].to_numpy()
x2 = sample_data_1216["x2 (% TA)"].to_numpy()
x3 = sample_data_1216["x3 (% 2 NDPA)"].to_numpy()
y = sample_data_1216["y (Specific Gravity)"].to_numpy()

# Nonlinear model
def model(X, b1, b2, b3):
    x1, x2, x3 = X
    return 1 / (b1*x1 + b2*x2 + b3*x3)

# Linear Model
def linear_model(X, b1, b2, b3):
    x1, x2, x3 = X
    return (b1*x1 + b2*x2 + b3*x3)

# STARTING VALUES from linear regression
z = 1/y
X_lin = np.column_stack([x1, x2, x3])
beta0 = np.linalg.lstsq(X_lin, z, rcond=None)[0]
