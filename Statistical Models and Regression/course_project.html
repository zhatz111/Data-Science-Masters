# Full model
def linear_regression(X, y):
    X = np.column_stack((np.ones(X.shape[0]), X))

    beta = np.linalg.pinv(X.T @ X) @ X.T @ y
    y_pred_full = X @ beta

    sst_full = np.sum((y - y_bar) ** 2)
    ssr_full = np.sum((y_pred_full - y_bar) ** 2)
    sse_full = np.sum((y - y_pred_full) ** 2)
    mse_full = sse_full / (len(y) - X.shape[1])

    r2_full = 1 - (sse_full / sst_full)
    adj_r2_full = 1 - (1 - r2_full) * (len(y) - 1) / (len(y) - X.shape[1] - 1)
    aic_full = len(y) * np.log(sse_full / len(y)) + 2 * X.shape[1]
    bic_full = len(y) * np.log(sse_full / len(y)) + np.log(len(y)) * X.shape[1]
    return (
        y_pred_full,
        sst_full,
        ssr_full,
        sse_full,
        mse_full,
        r2_full,
        adj_r2_full,
        aic_full,
        bic_full,
    )

def linear_regression_diagnostics(X, y):
    """
    Computes leverage, standardized residuals, and Cook's distance.

    Parameters
    ----------
    X : ndarray (n x p)
        Design matrix (without intercept column)
    y : ndarray (n,)
        Response vector

    Returns
    -------
    dict:
        'beta', 'fitted', 'residuals', 'leverage', 
        'std_residuals', 'cooks_distance'
    """

    # --------- 1. Add intercept column ---------
    n = X.shape[0]
    X_design = np.column_stack((np.ones(n), X))   # n x (p+1)
    
    # --------- 2. Fit OLS coefficients ---------
    XtX = X_design.T @ X_design
    XtX_inv = np.linalg.inv(XtX)
    beta = XtX_inv @ X_design.T @ y               # (p+1,)

    # --------- 3. Fitted values & residuals ---------
    y_hat = X_design @ beta
    residuals = y - y_hat

    # --------- 4. Estimate sigma^2 (MSE) ---------
    p = X_design.shape[1]
    mse = (residuals @ residuals) / (n - p)

    # --------- 5. Compute leverage h_i = diag(X (X'X)^(-1) X') ---------
    hat_matrix = X_design @ XtX_inv @ X_design.T
    leverage = np.diag(hat_matrix)

    # --------- 6. Standardized residuals ---------
    std_residuals = residuals / np.sqrt(mse * (1 - leverage))

    # --------- 7. Cook's Distance ---------
    cooks_distance = (std_residuals**2 * leverage) / (p * (1 - leverage))

    return {
        "beta": beta,
        "fitted": y_hat,
        "residuals": residuals,
        "leverage": leverage,
        "std_residuals": std_residuals,
        "cooks_distance": cooks_distance
    }

import numpy as np
from sklearn.model_selection import KFold
from sklearn.metrics import mean_squared_error, r2_score
from sklearn.linear_model import LinearRegression

def kfold_cv(X, y, k=5):
    kf = KFold(n_splits=k, shuffle=True, random_state=42)
    model = LinearRegression()

    rmse_scores = []
    r2_scores = []

    for train_idx, test_idx in kf.split(X):
        X_train, X_test = X[train_idx], X[test_idx]
        y_train, y_test = y[train_idx], y[test_idx]

        model.fit(X_train, y_train)
        y_pred = model.predict(X_test)

        rmse = np.sqrt(mean_squared_error(y_test, y_pred))
        rmse_scores.append(rmse)

        r2 = r2_score(y_test, y_pred)
        r2_scores.append(r2)

    return {
        "rmse_scores": rmse_scores,
        "mean_rmse": np.mean(rmse_scores),
        "std_rmse": np.std(rmse_scores),
        "r2_scores": r2_scores,
        "mean_r2": np.mean(r2_scores)
    }

results = kfold_cv(X, y, k=5)
print("K-Fold Cross Validation Results:\n")
for key, value in results.items():
    print(f"{key}: {value}")

from sklearn.model_selection import train_test_split

def train_test_validation(X, y, test_size=0.2):
    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=test_size, random_state=42
    )

    model = LinearRegression()
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)

    rmse = np.sqrt(mean_squared_error(y_test, y_pred))
    r2 = model.score(X_test, y_test)

    return {
        "rmse": rmse,
        "r2": r2,

def bootstrap_validation(X, y, B=1000):
    n = len(y)
    model = LinearRegression()

    coef_samples = []
    rmse_samples = []
    r2_samples = []

    for _ in range(B):
        sample_idx = np.random.choice(n, n, replace=True)
        oob_idx = np.setdiff1d(np.arange(n), sample_idx)

        if len(oob_idx) == 0:
            continue

        X_train, y_train = X[sample_idx], y[sample_idx]
        X_test, y_test = X[oob_idx], y[oob_idx]

        model.fit(X_train, y_train)
        y_pred = model.predict(X_test)

        coef_samples.append(model.coef_)
        rmse_samples.append(np.sqrt(mean_squared_error(y_test, y_pred)))

        r2 = r2_score(y_test, y_pred)
        r2_samples.append(r2)

    return {
        "coef_mean": np.mean(coef_samples, axis=0),
        "coef_std": np.std(coef_samples, axis=0),
        "rmse_mean": np.mean(rmse_samples),
        "rmse_std": np.std(rmse_samples),
        "r2_mean": np.mean(r2_samples),
        "r2_std": np.std(r2_samples)
    }

results_bootstrap = bootstrap_validation(X, y, B=10000)
print("Bootstrap Validation Results:\n")
for key, value in results_bootstrap.items():
    print(f"{key}: {value}")
